        function generatePDF() {

            const date = document.getElementById('date').value;

            const time = document.getElementById('time').value;

            const venue = document.getElementById('venue').value;

            const topics = document.getElementById('topics').value;

            const actions = document.getElementById('actions').value;


            const topicsList = topics.split('\n').filter(Boolean).map(item => `<li>${item}</li>`).join('');

            const actionsList = actions.split('\n').filter(Boolean).map(item => `<li>${item}</li>`).join('');


            const content = `

                <div class="pdf-container">

                    <div class="pdf-header">

                        <h2>Meeting Notes</h2>

                    </div>

                    <div class="pdf-section">

                        <h3>Meeting Details</h3>

                        <div class="pdf-content">

                            <p><strong>Date:</strong> ${date}</p>

                            <p><strong>Time:</strong> ${time}</p>

                            <p><strong>Venue:</strong> ${venue}</p>

                        </div>

                    </div>

                    <div class="pdf-section">

                        <h3>Topics of Discussion</h3>

                        <div class="pdf-content">

                            <ul>

                                ${topicsList || '<li>No topics recorded.</li>'}

                            </ul>

                        </div>

                    </div>

                    <div class="pdf-section">

                        <h3>Action Items</h3>

                        <div class="pdf-content">

                            <ul>

                                ${actionsList || '<li>No action items recorded.</li>'}

                            </ul>

                        </div>

                    </div>

                </div>

            `;


            const element = document.createElement('div');

            element.innerHTML = content;


            html2pdf().from(element).set({

                margin: 10,

                filename: `Meeting_Notes_${date.replace(/-/g, '_')}.pdf`,

                image: { type: 'jpeg', quality: 0.98 },

                html2canvas: { scale: 2, logging: true, dpi: 192, letterRendering: true },

                jsPDF: { unit: 'mm', format: 'a4', orientation: 'portrait' }

            }).save();

        }


        function getNotesText() {

            const date = document.getElementById('date').value;

            const time = document.getElementById('time').value;

            const venue = document.getElementById('venue').value;

            const topics = document.getElementById('topics').value;

            const actions = document.getElementById('actions').value;


            return `*Meeting Notes*

Date: ${date}

Time: ${time}

Venue: ${venue}


*Topics of Discussion:*

${topics}


*Action Items:*

${actions}`;

        }


        function shareWhatsApp() {

            const notes = getNotesText();

            const encodedText = encodeURIComponent(notes);

            window.open(`https://wa.me/?text=${encodedText}`);

        }


        function shareEmail() {

            const notes = getNotesText();

            const subject = encodeURIComponent('Meeting Notes');

            const body = encodeURIComponent(notes);

            window.location.href = `mailto:?subject=${subject}&body=${body}`;

        }
